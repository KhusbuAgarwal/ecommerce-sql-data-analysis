

-- Table Setup
CREATE TABLE products (
    product_id INT PRIMARY KEY,
    product_name VARCHAR(50),
    category VARCHAR(50),
    price DECIMAL(10,2)
);

CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    product_id INT,
    customer_name VARCHAR(50),
    order_date DATE
);

-- Data Extraction and Sorting
SELECT * FROM products 
WHERE price > 100 
ORDER BY price DESC;

-- Multi-Table Analysis using Joins
SELECT o.customer_name, p.product_name, p.price, o.order_date
FROM orders o
INNER JOIN products p ON o.product_id = p.product_id;

-- Aggregation and Category Summaries
SELECT category, COUNT(*) AS item_count, AVG(price) AS average_category_price
FROM products
GROUP BY category;

-- Subquery for Outlier Analysis (Prices above average)
SELECT product_name, price 
FROM products 
WHERE price > (SELECT AVG(price) FROM products);

-- View Creation for Reporting
CREATE VIEW high_value_inventory AS
SELECT * FROM products WHERE price > 500;

-- Testing the View
SELECT * FROM high_value_inventory;

-- Database Optimization: Indexing
CREATE INDEX idx_product_name ON products(product_name);
